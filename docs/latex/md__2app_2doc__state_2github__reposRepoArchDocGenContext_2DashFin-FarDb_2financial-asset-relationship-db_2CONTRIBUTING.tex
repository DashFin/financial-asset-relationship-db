\begin{DoxyItemize}
\item Keep branches short-\/lived (ideally \texorpdfstring{$<$}{<} 2 weeks)

\item Regularly sync with main to avoid conflicts
\end{DoxyItemize}

\begin{DoxyItemize}

\end{DoxyItemize}
